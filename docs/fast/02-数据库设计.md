## 2. 详细表结构

### 2.1 表格填写任务表 `lingdoc_form_task`

```sql
DROP TABLE IF EXISTS `lingdoc_form_task`;
CREATE TABLE `lingdoc_form_task` (
  `task_id`           varchar(64)   NOT NULL COMMENT '任务ID（UUID）',
  `user_id`           bigint(20)    NOT NULL COMMENT '用户ID（sys_user.user_id）',
  `task_name`         varchar(256)  NOT NULL COMMENT '任务名称（如：国家奖学金申请表）',
  `original_file_id`  varchar(64)   NOT NULL COMMENT '原始空白文档文件ID',
  `original_file_url` varchar(512)  NOT NULL COMMENT '原始文件存储路径（相对或绝对）',
  `original_file_name`varchar(256)  NOT NULL COMMENT '原始文件名',
  `filled_file_id`    varchar(64)   DEFAULT NULL COMMENT 'AI填写后文档文件ID',
  `filled_file_url`   varchar(512)  DEFAULT NULL COMMENT '填写后文件存储路径',
  `filled_file_name`  varchar(256)  DEFAULT NULL COMMENT '填写后文件名',
  `status`            char(1)       DEFAULT '0' COMMENT '状态：0待上传 1识别中 2待确认 3已生成 4失败',
  `ai_result`         longtext      DEFAULT NULL COMMENT 'AI原始识别结果JSON',
  `field_count`       int           DEFAULT 0 COMMENT '字段总数',
  `confirmed_count`   int           DEFAULT 0 COMMENT '用户已确认字段数',
  `token_cost`        int           DEFAULT 0 COMMENT 'Token消耗总量',
  `error_msg`         varchar(512)  DEFAULT NULL COMMENT '错误信息',
  `create_by`         varchar(64)   DEFAULT '' COMMENT '创建者',
  `create_time`       datetime      DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_by`         varchar(64)   DEFAULT '' COMMENT '更新者',
  `update_time`       datetime      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `remark`            varchar(500)  DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`task_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_status` (`status`),
  KEY `idx_create_time` (`create_time`),
  KEY `idx_user_status` (`user_id`, `status`),
  FULLTEXT KEY `ft_task_name` (`task_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='表格填写任务表';
```

**索引设计说明**：
- `PRIMARY KEY (task_id)`：主键查询
- `idx_user_id`：用户查询自己的任务列表
- `idx_status`：按状态筛选（如只查"待确认"）
- `idx_create_time`：按时间排序
- `idx_user_status`：联合索引，覆盖最常用的"查询某用户某状态的任务"场景
- `ft_task_name`：全文索引，支持任务名称搜索

### 2.2 表格字段表 `lingdoc_form_field`

```sql
DROP TABLE IF EXISTS `lingdoc_form_field`;
CREATE TABLE `lingdoc_form_field` (
  `field_id`      varchar(64)   NOT NULL COMMENT '字段ID（UUID）',
  `task_id`       varchar(64)   NOT NULL COMMENT '所属任务ID',
  `field_name`    varchar(128)  NOT NULL COMMENT '字段名称（如：姓名）',
  `field_type`    varchar(32)   DEFAULT 'text' COMMENT '字段类型：text/date/number/select',
  `field_label`   varchar(256)  DEFAULT NULL COMMENT '字段在文档中的原始标签文本',
  `ai_value`      varchar(512)  DEFAULT NULL COMMENT 'AI建议的填写值',
  `user_value`    varchar(512)  DEFAULT NULL COMMENT '用户最终确认的值',
  `is_confirmed`  char(1)       DEFAULT '0' COMMENT '是否已确认：0否 1是',
  `confidence`    decimal(3,2)  DEFAULT 0.00 COMMENT 'AI置信度（0.00~1.00）',
  `source_doc_id` varchar(64)   DEFAULT NULL COMMENT '值来源的统一文件ID（lingdoc_file_index.file_id）',
  `source_doc_name`varchar(256) DEFAULT NULL COMMENT '来源文档名称',
  `sort_order`    int           DEFAULT 0 COMMENT '字段在表格中的排序号',
  `create_time`   datetime      DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time`   datetime      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`field_id`),
  KEY `idx_task_id` (`task_id`),
  KEY `idx_field_name` (`field_name`),
  KEY `idx_confirmed` (`task_id`, `is_confirmed`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='表格字段表';
```

**索引设计说明**：
- `idx_task_id`：查询某任务的所有字段
- `idx_field_name`：按字段名搜索
- `idx_confirmed`：查询某任务中已确认/未确认的字段

### 2.3 任务参考文档关联表 `lingdoc_form_reference`

```sql
DROP TABLE IF EXISTS `lingdoc_form_reference`;
CREATE TABLE `lingdoc_form_reference` (
  `ref_id`        varchar(64)   NOT NULL COMMENT '关联ID（UUID）',
  `task_id`       varchar(64)   NOT NULL COMMENT '任务ID',
  `doc_id`        varchar(64)   NOT NULL COMMENT '文件ID（lingdoc_file_index.file_id）',
  `doc_name`      varchar(256)  NOT NULL COMMENT '文档名称',
  `doc_path`      varchar(512)  NOT NULL COMMENT '文档在Vault中的存储路径',
  `doc_type`      varchar(32)   DEFAULT NULL COMMENT '文档类型（pdf/docx/xlsx/doc/xls/txt/md/csv/json/xml/yaml/png/jpg等）',
  `relevance`     decimal(3,2)  DEFAULT 0.00 COMMENT 'AI相关性评分（0.00~1.00）',
  `is_selected`   char(1)       DEFAULT '1' COMMENT '是否被用户选中：0否 1是',
  `create_time`   datetime      DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`ref_id`),
  KEY `idx_task_id` (`task_id`),
  KEY `idx_doc_id` (`doc_id`),
  UNIQUE KEY `uk_task_doc` (`task_id`, `doc_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='任务参考文档关联表';
```

**索引设计说明**：
- `idx_task_id`：查询某任务的参考文档
- `idx_doc_id`：查询某文档被哪些任务引用
- `uk_task_doc`：唯一索引，防止同一文档重复关联到同一任务

## 5. 索引优化建议

### 5.1 初期索引（已包含在建表语句中）
已完成所有必要索引的创建，满足 MVP 阶段查询需求。

### 5.2 后续监控与优化
上线后通过 `EXPLAIN` 和慢查询日志监控以下高频 SQL：

```sql
-- 高频查询1：用户任务列表（需覆盖索引）
SELECT * FROM lingdoc_form_task 
WHERE user_id = ? AND status = ? 
ORDER BY create_time DESC LIMIT ?, ?;

-- 高频查询2：任务字段详情
SELECT * FROM lingdoc_form_field 
WHERE task_id = ? ORDER BY sort_order;

-- 高频查询3：任务参考文档
SELECT * FROM lingdoc_form_reference 
WHERE task_id = ? AND is_selected = '1';
```

若出现性能瓶颈，可考虑：
- 为 `lingdoc_form_task` 增加覆盖索引 `(user_id, status, create_time, task_id, task_name)`
- 为 `lingdoc_form_field` 增加覆盖索引 `(task_id, sort_order, field_id, field_name, ai_value, user_value)`
