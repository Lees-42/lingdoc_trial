**文档编号**: LingDoc-FAST-003  
**版本**: v2.0  
**基础路径**: `/lingdoc/form`  
**鉴权**: JWT Bearer Token

## 0. Vault 上下文传递（v2.0 新增）

自 2026-04-22 引入多Vault架构后，所有 `/lingdoc/` 前缀的接口（表格助手、文件管理、标签等）均需携带当前 Vault 上下文，后端据此路由到对应的 SQLite 数据源。

### 0.1 前端自动注入

`src/utils/request.js` 请求拦截器已自动处理：

```javascript
// 自动携带当前 Vault 路径（lingdoc 模块请求）
if (config.url && config.url.startsWith('/lingdoc/')) {
  const vaultStore = useVaultStore()
  if (vaultStore.currentVaultPath) {
    config.headers['X-Vault-Path'] = vaultStore.currentVaultPath
  }
}
```

- 前端通过 `vaultStore.currentVaultPath`（来自 `localStorage` 持久化）获取当前仓库路径
- 所有 `/lingdoc/` 请求自动在 Header 中附加 `X-Vault-Path`
- 若用户未选择仓库，`vaultStore` 会自动加载默认仓库

### 0.2 后端处理流程

```
HTTP Request (X-Vault-Path: D:/LingDocVault)
        ↓
VaultContextInterceptor → 提取 Header 写入 VaultContextHolder (ThreadLocal)
        ↓
VaultDataSourceAspect → AOP拦截，根据 VaultContextHolder 获取对应 SQLite DataSource
        ↓
DynamicDataSource → 切换数据源到 {vault_root}/.lingdoc/vault.db
        ↓
Controller → Service → Mapper → SQLite
```

### 0.3 无仓库场景

- 新用户首次登录时，若 `lingdoc_user_repo` 中无记录，前端 `VaultRepoManager` 组件强制弹出创建仓库弹窗
- 未选择仓库时调用 `/lingdoc/` 接口，后端返回 `500` 错误提示"未选择Vault"

### 2.1 上传表格文件

```
POST /lingdoc/form/upload
Content-Type: multipart/form-data
Authorization: Bearer {token}
```

#### Request

| 参数名 | 类型 | 必填 | 说明 |
|--------|------|------|------|
| `file` | MultipartFile | 是 | 表格文件（PDF/Word/Excel） |

## 3. 标签系统接口详情（v2.0 新增）

**响应**：
```json

  "code": 200,
  "msg": "操作成功",
  "data": [
    {
      "tagId": "tag_001",
      "tagName": "奖学金",
      "tagColor": "#67C23A",
      "tagScope": "A",

    }
  ]
}
```

### 3.2 新建标签

```
POST /lingdoc/tag
Content-Type: application/json
Authorization: Bearer {token}
X-Vault-Path: D:/LingDocVault

{ "tagName": "操作系统", "tagColor": "#409EFF", "tagScope": "A" }
```

### 3.3 获取文件标签（含继承）

```
GET /lingdoc/tag/file/{fileId}
Authorization: Bearer {token}
X-Vault-Path: D:/LingDocVault
```

**响应**：
```json
{
  "code": 200,
  "msg": "操作成功",
  "data": {
    "directTags": [
      { "bindingId": "bind_001", "tagId": "tag_001", "tagName": "奖学金", "tagColor": "#67C23A", "bindType": "0" }
    ],
    "inheritedTags": [
      { "tagId": "tag_002", "tagName": "学习资料", "tagColor": "#E6A23C", "bindType": "1" }
    ]
  }
}
```

### 3.4 绑定标签

```
POST /lingdoc/tag/bind
Content-Type: application/json
Authorization: Bearer {token}
X-Vault-Path: D:/LingDocVault

{
  "targetType": "F",
  "targetId": "f_abc123",
  "tagId": "tag_001"
}
```

**参数说明**：
- `targetType`: `F` = 文件, `D` = 目录
- `targetId`: 文件时为 `file_id`，目录时为目录路径（如 `学习资料/大三上`）
- `tagId`: 标签ID

**业务规则**：
- 幂等绑定：同一目标+同一标签的直接绑定不会重复创建
- 目录标签绑定后，其下所有文件自动继承该标签（`bind_type = '1'`）

### 3.5 解绑标签

```
DELETE /lingdoc/tag/bind/{bindingId}
Authorization: Bearer {token}
X-Vault-Path: D:/LingDocVault
```

**业务规则**：
- 只能解绑 `bind_type = '0'` 的直接绑定
- 继承标签（`bind_type = '1'`）不可直接解绑，需解绑父目录的直接绑定

## 4. 前端 API 封装

### 4.1 表格助手 API

**文件**: `src/api/lingdoc/form.js`

```javascript
import request from '@/utils/request'

// ==================== 任务管理 ====================

// 查询表格填写任务列表
export function listForm(query) {
  return request({
    url: '/lingdoc/form/list',
    method: 'get',
    params: query
  })
}

// 获取任务详情
export function getFormTask(taskId) {
  return request({
    url: '/lingdoc/form/' + taskId,
    method: 'get'
  })
}

// 上传表格文件
export function uploadForm(data) {
  return request({
    url: '/lingdoc/form/upload',
    method: 'post',
    data: data,
    headers: {
      'Content-Type': 'multipart/form-data'
    },
    timeout: 60000
  })
}

// 修改任务
export function updateForm(data) {
  return request({
    url: '/lingdoc/form',
    method: 'put',
    data: data
  })
}

// 删除任务
export function delForm(taskId) {
  return request({
    url: '/lingdoc/form/' + taskId,
    method: 'delete'
  })
}

// 批量删除
export function batchDelForm(taskIds) {
  return request({
    url: '/lingdoc/form/batch',
    method: 'delete',
    data: taskIds
  })
}

// ==================== 字段管理 ====================

// 批量更新字段值
export function updateFormFields(data) {
  return request({
    url: '/lingdoc/form/fields',
    method: 'put',
    data: data
  })
}

// ==================== 生成与下载 ====================

// 触发AI生成
export function generateForm(data) {
  return request({
    url: '/lingdoc/form/generate',
    method: 'post',
    data: data,
    timeout: 120000
  })
}

// 下载填写后文档
export function downloadForm(taskId) {
  return request({
    url: '/lingdoc/form/download/' + taskId,
    method: 'get',
    responseType: 'blob'
  })
}

// ==================== 参考文档 ====================

// 获取参考文档列表
export function getFormReferences(taskId) {
  return request({
    url: '/lingdoc/form/references/' + taskId,
    method: 'get'
  })
}
```

## 4. 统一响应格式

所有接口统一返回 `AjaxResult`：

```json
{
  "code": 200,        // 200成功，其他为错误码
  "msg": "操作成功",   // 提示信息
  "data": { }         // 业务数据
}
```

### 错误码定义

| code | 说明 |
|------|------|
| 200 | 成功 |
| 500 | 服务器内部错误 |
| 1001 | 参数校验失败 |
| 1002 | 文件上传失败 |
| 1003 | 文件格式不支持（仅支持 PDF/Word/Excel） |
| 1004 | 任务不存在 |
| 1005 | 无权限操作该任务 |
| 1006 | AI 服务调用失败 |
| 1007 | 文档生成失败 |
